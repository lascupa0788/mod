	on_state_control_changed = {
		effect = {

					limit = {
						NOT = {
							AND = {

							}
						}
					}

				}

					limit = {
						AND = {
							is_owned_by = PREV
							PREV = { ROOT = { has_war_with = FROM } }
						}
					}
